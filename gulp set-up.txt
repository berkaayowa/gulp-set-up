gulp.task('scripts', function(){
	gulp.src('js/*.js')
	.pipe(uglify())
	.pipe(gulp.dest('frontend-js/'));
});

gulp.task('watch', function(){
	gulp.watch('js/*.js', ['scripts']);
	gulp.watch('scss/*.scss', ['styles']);
});

gulp.task('default', ['scripts', 'styles', 'watch']);
